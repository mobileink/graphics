\begin{animateinline}[poster=first,%
    %% step,
    controls,
    begin={%
      \begin{tikzpicture}[domain=-\xmax:\xmax,scale=1]%
        \useasboundingbox[draw=black] (-\xmax,-\ymax)
        rectangle (\xmax,\ymax);%
    },
    end={\end{tikzpicture}%
    }%
  ]%
  {\FrameRate}%
  %% \draw[->] (-\xmax,0) -- (\xmax,0) node[right] {$x$}; %% x axis

  %% \draw[color=blue] plot (\x,0.12*\x) node[above right] {$0.12$};
  \multiframe{\FrameCount}{iCurrFrame=1+1}{%
    \draw[->] (-\xmax,0) -- (\xmax,0) node[right] {$x$}; %% x axis
    \draw[->] (0,-\ymax) -- (0,\ymax) node[above] {$y$}; %% y axis
    \pgfmathsetmacro{\pct}{\iCurrFrame/\FrameCount};
    \pgfmathsetmacro{\dist}{0.25-0.12};
    \pgfmathsetmacro{\delta}{0.12+(\pct*\dist)};
    \draw[color=blue] plot (\x,\delta*\x) node[above right] {$\iCurrFrame$};
  }
  \newframe
  \draw[->] (-\xmax,0) -- (\xmax,0) node[right] {$x$}; %% x axis
  \draw[->] (0,-\ymax) -- (0,\ymax) node[above] {$y$}; %% y axis
  \draw[color=blue] plot (\x,0.25*\x) node[above right] {$0.25$};
  \multiframe{\FrameCount}{iCurrFrame=1+1}{%
    \draw[->] (-\xmax,0) -- (\xmax,0) node[right] {$x$}; %% x axis
    \draw[->] (0,-\ymax) -- (0,\ymax) node[above] {$y$}; %% y axis
    \pgfmathsetmacro{\pct}{\iCurrFrame/\FrameCount};
    \pgfmathsetmacro{\dist}{0.11-0.25};
    \pgfmathsetmacro{\delta}{0.25+(\pct*\dist)};
    \draw[color=blue] plot (\x,\delta*\x) node[above right] {$\iCurrFrame$};
  }
\end{animateinline}
